<script>

// Name system
let userName = new URLSearchParams(window.location.search).get('name') || "Cadet";

document.addEventListener("DOMContentLoaded", () => {
    document.getElementById("q1text").innerText =
        `${userName}, do you like parties?`;
});

// Navigation
function nextPage(num){
    document.querySelectorAll(".page").forEach(p => p.classList.remove("active"));
    document.getElementById("p"+num).classList.add("active");

    let clickSound = document.getElementById("successSound");
    clickSound.currentTime = 0;
    clickSound.play().catch(()=>{});

    if(num === 4){
        document.getElementById("partySound").play().catch(()=>{});
        launchConfetti();

        document.getElementById("finalText").innerText =
            `Expecting you, ${userName}!!! 🎉🫡`;
    }
}

// Errors
function errorMsg(){
    let err = document.getElementById("errorSound");
    err.currentTime = 0;
    err.play().catch(()=>{});
    alert("A good cadet should always want a party");
}

function errorMsg2(){
    let err = document.getElementById("errorSound");
    err.currentTime = 0;
    err.play().catch(()=>{});
    alert("Incorrect answer 🤡");
}

// Moving button
function moveBtn(btn){
    btn.style.position = "absolute";
    btn.style.top = Math.random()*80 + "%";
    btn.style.left = Math.random()*80 + "%";
}

// Confetti (SIDE BURST FIXED)
function launchConfetti() {
    const duration = 2000;
    const end = Date.now() + duration;

    (function frame() {

        confetti({
            particleCount: 5,
            angle: 60,
            spread: 55,
            origin: { x: 0, y: 0.7 }
        });

        confetti({
            particleCount: 5,
            angle: 120,
            spread: 55,
            origin: { x: 1, y: 0.7 }
        });

        if (Date.now() < end) {
            requestAnimationFrame(frame);
        }
    })();
}
